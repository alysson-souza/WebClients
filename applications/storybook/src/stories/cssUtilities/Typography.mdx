import { useEffect, useState } from 'react';

import { Meta, Source } from '@storybook/addon-docs/blocks';

import { IcChevronUp } from '@proton/icons/icons/IcChevronUp';

<Meta title="CSS Utilities/Typography" />

export const BackToTop = () => {
    const [visible, setVisible] = useState(false);

    useEffect(() => {
        const onScroll = () => setVisible(window.scrollY > 500);
        onScroll();
        window.addEventListener('scroll', onScroll, { passive: true });
        return () => window.removeEventListener('scroll', onScroll);
    }, []);

    if (!visible) {
        return null;
    }

    return (
        <button
            type="button"
            aria-label="Back to top"
            className="inline-flex items-center gap-1"
            style={{
                position: 'fixed',
                right: '1.5rem',
                bottom: '1.5rem',
                zIndex: 1000,
                padding: '0.5rem 0.75rem',
                borderRadius: '2rem',
                border: '1px solid var(--border-norm)',
                background: 'var(--background-norm)',
                color: 'var(--text-norm)',
                boxShadow: '0 0.5rem 1rem rgba(0, 0, 0, 0.15)',
                cursor: 'pointer',
            }}
            onClick={() => window.scrollTo({ top: 0, behavior: 'smooth' })}
        >
            <IcChevronUp size={3} /> Back to top
        </button>
    );

};

## Overflow & truncation

### `text-ellipsis` with `text-pre`

If you want to apply <code>text-ellipsis</code> and <code>pre</code> classes, you need to wrap your content like this:

<div class="flex justify-space-between mb-8 flex-column md:flex-row">
  <div class="w-1/2 self-center">
    <span class="text-ellipsis inline-block max-w-1/3" ><span class="text-pre">Lorem    ipsum dolor sit amet</span></span><br />
    To be used with <code>max-w-full</code> and <code>inline-block</code> (or with flexbox helpers).
  </div>
  <div class="w-1/2">
<Source
  language='html'
  light
  format
  code={`
<span className="text-ellipsis">
  <span className="text-pre">
  Lorem    ipsum dolor sit amet
  </span>
  </span>
`}
/>

  </div>
</div>
